## 1. root layout 메타 교체 + lang ko (가장 먼저)

지금 `src/app/layout.tsx:17-20`는 `create-next-app`이 만들어준 기본값 그대로입니다. 이게 모든 페이지 제목으로 나가다 보니 검색엔진에 "Create Next App"으로 색인되고요. `lang`도 `en`이라, 한국어 페이지인데 영어로 선언돼 있습니다.

// src/app/layout.tsx

  metadataBase: new URL('https://tiki.example.com'),

## 2. [eventId] generateMetadata + sitemap/robots

### 2-1. 이벤트 상세 generateMetadata

지금은 모든 상세 페이지 제목이 똑같은데, 공연마다 제목·설명이 달라야 검색·공유에서 의미가 생깁니다. 같은 데이터를 메타와 본문에서 두 번 조회하지 않도록 `React.cache`로 감싸는 게 핵심입니다.

```tsx
// src/app/[eventId]/page.tsx
import { cache } from 'react';
import type { Metadata } from 'next';

// 메타와 본문이 같은 호출을 공유하도록 cache로 감쌉니다(중복 조회 방지)
const getEvent = cache(async (eventId: string) => {
  return getEventDetail(eventId); // 기존 queries.ts 함수 재사용
});

export async function generateMetadata(
  { params }: { params: Promise<{ eventId: string }> }
): Promise<Metadata> {
  const { eventId } = await params;
  const event = await getEvent(eventId);
  if (!event) return { title: '이벤트를 찾을 수 없습니다' };

  return {
    title: event.title, // template 덕분에 "공연명 | TiKi"가 됩니다
    description: event.description?.slice(0, 120) ?? undefined,
    openGraph: { title: event.title, images: event.posterUrl ? [event.posterUrl] : [] },
  };
}

export default async function EventDetailPage({ params }: { params: Promise<{ eventId: string }> }) {
  const { eventId } = await params;
  const event = await getEvent(eventId); // 위 generateMetadata와 같은 cache 공유
  // ...기존 렌더...
}
```

### 2-2. sitemap.ts

지금 `/sitemap.xml`이 500입니다. 파일이 없어서입니다. 공개된 이벤트 URL을 펼쳐줍니다.

```ts
// src/app/sitemap.ts
import type { MetadataRoute } from 'next';

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const base = 'https://tiki.example.com';
  // 공개 이벤트만 가져옵니다(비공개/숨김 제외)
  const events = await getPublicEventIds(); // [{ eventId, updatedAt }]

  const eventUrls = events.map((e) => ({
    url: `${base}/${e.eventId}`,
    lastModified: e.updatedAt,
  }));

  return [
    { url: base, lastModified: new Date() },
    { url: `${base}/search`, lastModified: new Date() },
    ...eventUrls,
  ];
}
```

## 3. cancelReservation await (리디렉션 에러 경로)

`PaymentForm.tsx:155` 데스크탑 경로는 이미 `await cancelReservation(orderId)`로 잘 돼 있습니다. 문제는 리디렉션 에러 경로(`:104`)입니다. `useEffect` 콜백은 동기 함수라 그 안에서 바로 `await`를 못 쓰니, 내부에 async IIFE를 하나 두고 그 안에서 await 하면 됩니다.

```tsx
// src/app/payment/[orderId]/_components/PaymentForm.tsx  (useEffect 내부)
useEffect(() => {
  if (!redirectedPaymentId) return;

  if (redirectErrorCode) {
    // 즉시실행 async 함수로 감싸 취소가 끝날 때까지 기다립니다
    (async () => {
      await cancelReservation(orderId); // 취소 완료 보장
      toast.error(searchParams.get('message') || '결제가 취소되었습니다.');
    })();
    return;
  }
  // ...confirmPayment 경로는 그대로...
}, []);
```

await 없이 두면 취소 요청이 떠나기만 하고 결과를 기다리지 않아, 실패 시 주문이 `ordered` 상태로 남아 예매목록·재고에 유령처럼 걸릴 수 있습니다. 데스크탑 경로와 동작을 맞춰주면 됩니다.

## 4. 좌석 점유 TTL 일치 (held_until 기준 해제)

지금 좌석은 `held_until = now() + 15분`으로 잡는데(`20260629120100_seat_order_rpcs.sql:53`), 청소 cron `cancel_stale_orders`는 `created_at < now() - 30분` 기준이라 정작 `held_until`이 쓰이질 않습니다. 15분 안내와 30분 실제 해제가 어긋나는 거죠.

가장 단순한 방법은, 만료된 `held` 좌석을 `held_until` 기준으로 직접 푸는 함수를 만들고 cron에 거는 겁니다.

```sql
-- supabase/migrations/2026xxxxxxxxxx_release_expired_seats.sql
create or replace function public.release_expired_seats()
returns int
language plpgsql
security definer
set search_path = public
as $$
declare
  v_count int;
begin
  -- held 상태이고 hold 시간이 지난 좌석을 해제. 결제 완료(paid)된 좌석은 건드리지 않습니다.
  with expired as (
    delete from order_seat
     where status = 'held'
       and held_until < now()
    returning order_id
  )
  select count(*) into v_count from expired;
  return v_count;
end;
$$;

-- 1분마다 실행(기존 cancel_stale_orders cron과 같은 방식)
select cron.schedule(
  'release-expired-seats',
  '* * * * *',
  $$ select public.release_expired_seats(); $$
);
```

이렇게 하면 15분 hold가 진짜로 15분 뒤에 풀려서, 안내 문구와 실제 동작이 맞습니다. (좌석 해제와 함께 그 주문 자체도 취소 처리하고 싶으면, `expired`의 `order_id`로 `orders.status = 'cancelled'`까지 묶어주면 됩니다.)

## 5. 좌석 수 서버 상한

`api/orders/seats/route.ts:20-28`에서 지금은 빈 배열만 막습니다. 상한을 한 줄 추가합니다.

```ts
const MAX_SEATS_PER_ORDER = 8; // 1회 예매 최대 좌석 수

if (
  !eventId || !slotId || !ticketGradeId ||
  !Array.isArray(seatIds) ||
  seatIds.length === 0 ||
  seatIds.length > MAX_SEATS_PER_ORDER  // 추가
) {
  return fail('invalid order payload', 400);
}
```
